  if (dato.class == Fraccion)
     n = dato.num * @num
     d = dato.den * @den

def +(dato)

     n = dato.num * @den + dato.den * @num
     d = @den * dato.den
     Fraccion.new(n,d)

  end

def <=>(fraccion)
   if (@num * fraccion.den < @den * fraccion.num)
         return -1
   end
   if (@num * fraccion.den > @den * fraccion.num)
         return 1
   end
   if(@num * fraccion.den == @den * fraccion.num)
         return 0
   end
end
